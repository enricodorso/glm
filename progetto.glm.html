# Dividi i dati in training e test set
set.seed(123)
trainIndex <- createDataPartition(data$Numero_utilizzi_piattaforme, p = 0.7, list = FALSE)
trainData <- data[trainIndex, ]
testData <- data[-trainIndex, ]

# Definisci le variabili
x_train <- model.matrix(Numero_utilizzi_piattaforme ~ ., data = trainData)[, -1]
y_train <- trainData$Numero_utilizzi_piattaforme

x_test <- model.matrix(Numero_utilizzi_piattaforme ~ ., data = testData)[, -1]
y_test <- testData$Numero_utilizzi_piattaforme

# Cross-validation per selezionare lambda
cv.fit <- cv.glmnet(x_train, y_train, family = "poisson", alpha = 1)
best_lambda <- cv.fit$lambda.min

# Selezione delle variabili
coefs <- coef(cv.fit, s = best_lambda)
active.vars <- which(coefs != 0)

# Modelli con le variabili selezionate
poisson_model <- glm(y_train ~ ., data = trainData[, c("Numero_utilizzi_piattaforme", names(coefs)[active.vars])], family = poisson)
zeroinfl_model <- zeroinfl(y_train ~ . | ., data = trainData[, c("Numero_utilizzi_piattaforme", names(coefs)[active.vars])])

# Predizioni sul set di test
poisson_pred <- predict(poisson_model, newdata = testData, type = "response")
zeroinfl_pred <- predict(zeroinfl_model, newdata = testData, type = "response")

# Valutazione della performance
# RMSE (Root Mean Squared Error)
rmse_poisson <- sqrt(mean((poisson_pred - y_test)^2))
rmse_zeroinfl <- sqrt(mean((zeroinfl_pred - y_test)^2))

# MAE (Mean Absolute Error)

mae_zeroinfl <- mean(abs(zeroinfl_pred - y_test))
